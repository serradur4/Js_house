const display1 = document.getElementById("display1");
const display2 = document.getElementById("display2");
const display3 = document.getElementById("display3");
const display4 = document.getElementById("display4");
const display5 = document.getElementById("display5");


function somma() {
    const input1 = parseInt(document.getElementById("addendo1").value);
    const input2 = parseInt(document.getElementById("addendo2").value);
    const sum = input1 + input2;
    display1.value = sum;
}

function sottrai() {
    const input1 = parseInt(document.getElementById("minuendo").value);
    const input2 = parseInt(document.getElementById("sottraendo").value);
    const sot = input1 - input2;
    display2.value = sot;
}

function moltiplica() {
    const input1 = parseInt(document.getElementById("fattore1").value);
    const input2 = parseInt(document.getElementById("fattore2").value);
    const pro = input1 * input2;
    display3.value = pro;
}

function dividi() {
    const input1 = parseInt(document.getElementById("dividendo").value);
    const input2 = parseInt(document.getElementById("divisore").value);
    const div = input1 / input2;
    display4.value = div;
}

function faipotenza() {
    const input1 = parseInt(document.getElementById("numero").value);
    const input2 = parseInt(document.getElementById("esponente").value);
    const pot = Math.pow(input1, input2);
    display5.value = pot;
}

function faifattoriale() {
    const input = parseInt(document.getElementById("fattoriale").value);
    let result = 1;
    for (let i = 1; i <= input; i++) {
        result *= i;
    }
    display6.value = result;
}
function clearDisplay(){
    document.getElementById("addendo1").value = "";
    document.getElementById("addendo2").value = "";
    document.getElementById("minuendo").value = "";
    document.getElementById("sottraendo").value = "";
    document.getElementById("fattore1").value = "";
    document.getElementById("fattore2").value = "";
    document.getElementById("dividendo").value = "";
    document.getElementById("divisore").value = "";
    document.getElementById("numero").value = "";
    document.getElementById("esponente").value = "";
    document.getElementById("fattoriale").value = "";
    display1.value = "";
    display2.value = "";
    display3.value = "";
    display4.value = "";
    display5.value = "";
    display6.value = "";
}
